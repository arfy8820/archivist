Now that the basic outline of the program is working again, with podcast-dl support, let's rewrite the command-line parser using argu. Here is the template I would like, based on code using argparse in python.

```python
def build_parser() -> argparse.ArgumentParser:
    parser = argparse.ArgumentParser(prog="yt-archive", description="Archive YouTube channels, playlists, and podcasts via yt-dlp.")
    parser.add_argument("--config", "-c", help="Path to config file. Defaults to the platform config directory.")
    parser.add_argument("--json", "-j", action="store_true", help="Emit JSON output.")
    parser.add_argument("--quiet", action="store_true", help="Reduce human-readable output where possible.")
    parser.add_argument("--version", "-v", action="version", version=f"%(prog)s version {VERSION} using yt-dlp {YTDLPVERSION}.")

    subparsers = parser.add_subparsers(dest="command", required=True)

    subparsers.add_parser("list", help="List configured targets.")

    config_parser = subparsers.add_parser("config", help="Show or set global configuration options.")
    config_subparsers = config_parser.add_subparsers(dest="action", help="The action to perform on the config")
    show_parser = config_subparsers.add_parser("show", help="Display global properties from the configuration")
    show_parser.add_argument("property", help="The proprety of the config to show, leave empty to show global properties, and summaries for targets and yt_dlp", nargs="?")
    set_parser = config_subparsers.add_parser("set", help="Set or clear a configuration option.")
    set_parser.add_argument("property", help="The configuration property to modify", choices=["base_dir", "podcast_dir", "default_output_template", "podcast_template", "targets", "yt_dlp"])
    set_parser.add_argument("value", help="The value of the property. Leave empty to clear, if valid.", nargs="?")

    probe_parser = subparsers.add_parser("probe", help="Probe a target and report detected mode.")
    probe_parser.add_argument("name")

    sync_parser = subparsers.add_parser("sync", help="Sync one target or all targets. Defaults to all targets.")
    sync_parser.add_argument("name", nargs="?", help="Optional target name. Omit to sync all targets.")
    sync_parser.add_argument("--all", action="store_true", help="Sync all configured targets.")

    add_parser = subparsers.add_parser("add", help="Add a target to the config.")
    add_parser.add_argument("--url", help="YouTube channel or playlist URL.")
    add_parser.add_argument("--label", help="Label used as the target name and default subdirectory.")
    add_parser.add_argument("--output", help="Optional per-target output template override.")

    remove_parser = subparsers.add_parser("remove", help="Remove a target from the config.")
    remove_parser.add_argument("name", help="Target name to remove.")
    remove_parser.add_argument("--delete-archive", action="store_true", help="Also delete the target's yt-dlp archive file if it exists.")
    
    return parser
```

This will allow for much ritcher config editing, among other things. and using a well-known parser system rather than a hand-rolled parser.
